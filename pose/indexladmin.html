生成算法

// 1. 获取机器ID（浏览器指纹） machineId = btoa(userAgent + language + screenSize + colorDepth + timezone) // 2. 计算哈希 data = machineId + "_" + SECRET_KEY + "_" + timestamp hash = simpleHash(data) // 返回8位16进制大写 // 3. 格式化 license = "520MUS-" + hash[0:4] + "-" + hash[4:8] + "-" + hash[8:12] + "-" + hash[12:16] // 4. simpleHash 实现 function simpleHash(str) { let hash = 0; for (let i = 0; i < str.length; i++) { hash = ((hash << 5) - hash) + str.charCodeAt(i); hash = hash & hash; } return Math.abs(hash).toString(16).toUpperCase().padStart(8, '0'); }
